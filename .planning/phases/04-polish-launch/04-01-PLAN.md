chatBubble.addEventListener('click', () => {
    if (chatWindow.classList.contains('active')) {
        // Close animation
        chatWindow.classList.remove('active');
        chatWindow.addEventListener('transitionend', function handler() {
            chatWindow.removeEventListener('transitionend', handler);
            chatWindow.classList.remove('transitioning');

Add a `showChips()` function:

```javascript
let chipsShown = false;

function showChips() {
    const chipsContainer = document.createElement('div');
    chipsContainer.className = 'chat-chips';
    chipsContainer.id = 'chat-chips';

    const chipTexts = [
        'Tell me about his book',
        'Basketball career',
        'How to contact him'
    ];

    chipTexts.forEach(text => {
        const chip = document.createElement('button');
        chip.className = 'chat-chip';
        chip.textContent = text;
        chip.addEventListener('click', () => {
            removeChips();
            chatInput.value = text;
            sendMessage();
        });
        chipsContainer.appendChild(chip);
    });

    chatMessages.appendChild(chipsContainer);
    chatMessages.scrollTop = chatMessages.scrollHeight;
    chipsShown = true;
}

function removeChips() {
    const chips = document.getElementById('chat-chips');
    if (chips) chips.remove();
}
```
